## Overview

This phase implements the dual authentication system for Stash:

1. **Spotify** -- OAuth 2.0 Authorization Code with PKCE via AppAuth library + Custom Chrome Tabs
2. **YouTube Music** -- Google TV device OAuth flow (limited-input device) + InnerTube API client

Both services share a unified `TokenManager` interface backed by AES-256-GCM encrypted storage (Tink + Android Keystore + Proto DataStore). The Settings screen provides connect/disconnect UI for both accounts.

---

## Architecture Decisions

| Decision | Choice | Rationale |
|----------|--------|-----------|
| Spotify OAuth library | AppAuth-Android (`net.openid:appauth:0.11.x`) | RFC 8252 compliant, handles PKCE automatically, Custom Tabs built-in, battle-tested |
| YouTube Music auth | Manual Google TV device flow via OkHttp | No library needed -- simple HTTP polling. ytmusicapi pattern. |
| Token encryption | Tink AES-256-GCM + Android Keystore master key | Spec requirement. Tink manages key rotation. Keystore protects the keyset. |
| Token persistence | Proto DataStore (encrypted via Tink streaming AEAD) | Type-safe, coroutine-native, better than EncryptedSharedPreferences (deprecated) |
| InnerTube HTTP client | OkHttp (not Ktor) | Matches spec tech stack. Consistent with rest of app. Less dependency overhead. |
| In-memory token cache | `MutableStateFlow<AccessToken?>` per service | Reactive, thread-safe, allows UI to observe auth state changes |

```kotlin
package com.stash.core.auth.crypto

import android.content.Context
import com.google.crypto.tink.Aead
import com.google.crypto.tink.KeyTemplates
import com.google.crypto.tink.aead.AeadConfig
import com.google.crypto.tink.integration.android.AndroidKeysetManager
import dagger.hilt.android.qualifiers.ApplicationContext
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class TinkEncryptionManager @Inject constructor(
    @ApplicationContext private val context: Context,
) {
    private val aead: Aead by lazy {
        AeadConfig.register()

        val keysetManager = AndroidKeysetManager.Builder()
            .withSharedPref(
                context,
                KEYSET_NAME,
                PREF_FILE_NAME,
            )
            .withKeyTemplate(KeyTemplates.get("AES256_GCM"))
            .withMasterKeyUri(MASTER_KEY_URI)
            .build()

        keysetManager.keysetHandle.getPrimitive(Aead::class.java)
    }

    fun encrypt(plaintext: ByteArray, associatedData: ByteArray = EMPTY_AD): ByteArray {
        return aead.encrypt(plaintext, associatedData)
    }

    fun decrypt(ciphertext: ByteArray, associatedData: ByteArray = EMPTY_AD): ByteArray {
        return aead.decrypt(ciphertext, associatedData)
    }

    companion object {
        private const val KEYSET_NAME = "stash_auth_keyset"
        private const val PREF_FILE_NAME = "stash_auth_keyset_prefs"
        private const val MASTER_KEY_URI = "android-keystore://stash_master_key"
        private val EMPTY_AD = ByteArray(0)
    }
}
```

#### Task B.3 -- Implement EncryptedAuthSerializer for Proto DataStore
**Time:** 5 min | **File:** `core/auth/src/main/java/com/stash/core/auth/crypto/EncryptedAuthSerializer.kt`

```kotlin
package com.stash.core.auth.crypto

import androidx.datastore.core.Serializer
import com.stash.core.auth.proto.AuthTokensProto
import java.io.InputStream
import java.io.OutputStream
import javax.inject.Inject

class EncryptedAuthSerializer @Inject constructor(
    private val encryption: TinkEncryptionManager,
) : Serializer<AuthTokensProto> {

    override val defaultValue: AuthTokensProto = AuthTokensProto.getDefaultInstance()

    override suspend fun readFrom(input: InputStream): AuthTokensProto {
        val encryptedBytes = input.readBytes()
        if (encryptedBytes.isEmpty()) return defaultValue

        return try {
            val decryptedBytes = encryption.decrypt(encryptedBytes)
            AuthTokensProto.parseFrom(decryptedBytes)
        } catch (e: Exception) {
            // If decryption fails (e.g., key rotation, corruption), return default
            // This forces re-authentication -- safe fallback
            defaultValue
        }
    }

    override suspend fun writeTo(t: AuthTokensProto, output: OutputStream) {
        val plainBytes = t.toByteArray()
        val encryptedBytes = encryption.encrypt(plainBytes)
        output.write(encryptedBytes)
    }
}
```

```kotlin
package com.stash.core.auth.store

import android.content.Context
import androidx.datastore.core.DataStore
import androidx.datastore.core.DataStoreFactory
import com.stash.core.auth.crypto.EncryptedAuthSerializer
import com.stash.core.auth.model.ServiceToken
import com.stash.core.auth.model.UserInfo
import com.stash.core.auth.proto.AuthTokensProto
import com.stash.core.auth.proto.SpotifyTokenProto
import com.stash.core.auth.proto.YouTubeTokenProto
import dagger.hilt.android.qualifiers.ApplicationContext
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.map
import java.io.File
import java.time.Instant
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class EncryptedTokenStore @Inject constructor(
    @ApplicationContext private val context: Context,
    serializer: EncryptedAuthSerializer,
) {
    private val dataStore: DataStore<AuthTokensProto> = DataStoreFactory.create(
        serializer = serializer,
        produceFile = { File(context.filesDir, "auth_tokens.pb") },
    )

    // --- Spotify ---

    val spotifyToken: Flow<ServiceToken?> = dataStore.data.map { proto ->
        proto.spotify.takeIf { it.refreshToken.isNotEmpty() }?.toServiceToken()
    }

    val spotifyUser: Flow<UserInfo?> = dataStore.data.map { proto ->
        proto.spotify.takeIf { it.userId.isNotEmpty() }?.toUserInfo()
    }

    suspend fun saveSpotifyToken(token: ServiceToken, user: UserInfo? = null) {
        dataStore.updateData { current ->
            current.toBuilder().setSpotify(
                current.spotify.toBuilder().apply {
                    accessToken = token.accessToken
                    refreshToken = token.refreshToken
                    expiresAtEpochSeconds = token.expiresAtEpoch
                    scope = token.scope
                    if (user != null) {
                        userId = user.id
                        displayName = user.displayName
                        profileImageUrl = user.imageUrl.orEmpty()
                    }
                }
            ).build()
        }
    }

    suspend fun clearSpotify() {
        dataStore.updateData { current ->
            current.toBuilder()
                .setSpotify(SpotifyTokenProto.getDefaultInstance())
                .build()
        }
    }

    // --- YouTube Music ---

    val youTubeToken: Flow<ServiceToken?> = dataStore.data.map { proto ->
        proto.youtube.takeIf { it.refreshToken.isNotEmpty() }?.toServiceToken()
    }

    val youTubeUser: Flow<UserInfo?> = dataStore.data.map { proto ->
        proto.youtube.takeIf { it.displayName.isNotEmpty() }?.toUserInfo()
    }

    suspend fun saveYouTubeToken(token: ServiceToken, user: UserInfo? = null) {
        dataStore.updateData { current ->
            current.toBuilder().setYoutube(
                current.youtube.toBuilder().apply {
                    accessToken = token.accessToken
                    refreshToken = token.refreshToken
                    expiresAtEpochSeconds = token.expiresAtEpoch
                    scope = token.scope
                    if (user != null) {
                        channelHandle = user.id
                        displayName = user.displayName
                        profileImageUrl = user.imageUrl.orEmpty()
                    }
                }
            ).build()
        }
    }

    suspend fun clearYouTube() {
        dataStore.updateData { current ->
            current.toBuilder()
                .setYoutube(YouTubeTokenProto.getDefaultInstance())
                .build()
        }
    }

    // --- Mapping helpers ---

    private fun SpotifyTokenProto.toServiceToken() = ServiceToken(
        accessToken = accessToken,
        refreshToken = refreshToken,
        expiresAtEpoch = expiresAtEpochSeconds,
        scope = scope,
    )

    private fun SpotifyTokenProto.toUserInfo() = UserInfo(
        id = userId,
        displayName = displayName,
        imageUrl = profileImageUrl.ifEmpty { null },
    )

    private fun YouTubeTokenProto.toServiceToken() = ServiceToken(
        accessToken = accessToken,
        refreshToken = refreshToken,
        expiresAtEpoch = expiresAtEpochSeconds,
        scope = scope,
    )

    private fun YouTubeTokenProto.toUserInfo() = UserInfo(
        id = channelHandle,
        displayName = displayName,
        imageUrl = profileImageUrl.ifEmpty { null },
    )
}
```

```kotlin
package com.stash.core.auth.model

import java.time.Instant

data class ServiceToken(
    val accessToken: String,
    val refreshToken: String,
    val expiresAtEpoch: Long,
    val scope: String,
) {
    val isExpired: Boolean
        get() = Instant.now().epochSecond >= expiresAtEpoch

    val isExpiringSoon: Boolean
        get() = Instant.now().epochSecond >= expiresAtEpoch - REFRESH_BUFFER_SECONDS

    companion object {
        const val REFRESH_BUFFER_SECONDS = 300L // 5 minutes
    }
}

data class UserInfo(
    val id: String,
    val displayName: String,
    val imageUrl: String?,
)

enum class AuthService {
    SPOTIFY,
    YOUTUBE_MUSIC,
}

sealed class AuthState {
    data object NotConnected : AuthState()
    data object Connecting : AuthState()
    data class Connected(val user: UserInfo) : AuthState()
    data class Error(val message: String, val isRecoverable: Boolean) : AuthState()
}

/** YouTube device flow state while user is entering the code */
data class DeviceCodeState(
    val deviceCode: String,
    val userCode: String,
    val verificationUrl: String,
    val expiresAtEpoch: Long,
    val intervalSeconds: Int,
)

sealed class AuthError : Exception() {
    data class TokenRefreshFailed(val service: AuthService, override val cause: Throwable?) : AuthError()
    data class AuthorizationDenied(val service: AuthService) : AuthError()
    data class DeviceCodeExpired(val service: AuthService) : AuthError()
    data class NetworkError(override val cause: Throwable?) : AuthError()
    data class InvalidToken(val service: AuthService) : AuthError()
}
```

```kotlin
package com.stash.core.auth

import com.stash.core.auth.model.AuthService
import com.stash.core.auth.model.AuthState
import kotlinx.coroutines.flow.StateFlow

/**
 * Central token management for all authenticated services.
 *
 * Responsibilities:
 * - In-memory caching of access tokens (never persists access tokens to disk)
 * - Proactive refresh 5 minutes before expiry
 * - Thread-safe concurrent access via coroutines
 * - Emits [AuthState] for UI observation
 */
interface TokenManager {

    /** Current auth state for Spotify. Observe to update UI. */
    val spotifyAuthState: StateFlow<AuthState>

    /** Current auth state for YouTube Music. Observe to update UI. */
    val youTubeAuthState: StateFlow<AuthState>

    /**
     * Returns a valid access token for the given service.
     * Refreshes automatically if expired or expiring soon.
     * Throws [com.stash.core.auth.model.AuthError.TokenRefreshFailed] if refresh fails.
     * Throws [com.stash.core.auth.model.AuthError.InvalidToken] if no token exists.
     */
    suspend fun getAccessToken(service: AuthService): String

    /**
     * Store tokens after successful OAuth flow completion.
     * Called by the OAuth flow handlers after receiving tokens.
     */
    suspend fun onTokensReceived(
        service: AuthService,
        accessToken: String,
        refreshToken: String,
        expiresInSeconds: Long,
        scope: String,
    )

    /** Remove all tokens for a service. Revoke if possible. */
    suspend fun disconnect(service: AuthService)

    /** Initialize on app start -- load refresh tokens from encrypted store. */
    suspend fun initialize()
}
```

#### Task C.2 -- Implement TokenManagerImpl (core logic)
**Time:** 5 min | **File:** `core/auth/src/main/java/com/stash/core/auth/TokenManagerImpl.kt`

```kotlin
package com.stash.core.auth

import com.stash.core.auth.model.AuthError
import com.stash.core.auth.model.AuthService
import com.stash.core.auth.model.AuthState
import com.stash.core.auth.model.ServiceToken
import com.stash.core.auth.model.UserInfo
import com.stash.core.auth.refresh.SpotifyTokenRefresher
import com.stash.core.auth.refresh.YouTubeTokenRefresher
import com.stash.core.auth.store.EncryptedTokenStore
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.flow.first
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock
import java.time.Instant
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class TokenManagerImpl @Inject constructor(
    private val tokenStore: EncryptedTokenStore,
    private val spotifyRefresher: SpotifyTokenRefresher,
    private val youTubeRefresher: YouTubeTokenRefresher,
) : TokenManager {

    // In-memory token cache -- access tokens NEVER persisted to disk
    private var spotifyTokenCache: ServiceToken? = null
    private var youTubeTokenCache: ServiceToken? = null

    // Mutexes prevent concurrent refresh races
    private val spotifyMutex = Mutex()
    private val youTubeMutex = Mutex()

    private val _spotifyAuthState = MutableStateFlow<AuthState>(AuthState.NotConnected)
    override val spotifyAuthState: StateFlow<AuthState> = _spotifyAuthState.asStateFlow()

    private val _youTubeAuthState = MutableStateFlow<AuthState>(AuthState.NotConnected)
    override val youTubeAuthState: StateFlow<AuthState> = _youTubeAuthState.asStateFlow()

    override suspend fun initialize() {
        // Load persisted refresh tokens + user info from encrypted store
        val spotifyToken = tokenStore.spotifyToken.first()
        val spotifyUser = tokenStore.spotifyUser.first()
        if (spotifyToken != null && spotifyUser != null) {
            spotifyTokenCache = spotifyToken
            _spotifyAuthState.value = AuthState.Connected(spotifyUser)
        }

        val ytToken = tokenStore.youTubeToken.first()
        val ytUser = tokenStore.youTubeUser.first()
        if (ytToken != null && ytUser != null) {
            youTubeTokenCache = ytToken
            _youTubeAuthState.value = AuthState.Connected(ytUser)
        }
    }

    override suspend fun getAccessToken(service: AuthService): String {
        return when (service) {
            AuthService.SPOTIFY -> getSpotifyAccessToken()
            AuthService.YOUTUBE_MUSIC -> getYouTubeAccessToken()
        }
    }

    private suspend fun getSpotifyAccessToken(): String = spotifyMutex.withLock {
        val cached = spotifyTokenCache
            ?: throw AuthError.InvalidToken(AuthService.SPOTIFY)

        if (!cached.isExpiringSoon) return cached.accessToken

        return try {
            val refreshed = spotifyRefresher.refresh(cached.refreshToken)
            val newToken = cached.copy(
                accessToken = refreshed.accessToken,
                // Spotify may return a new refresh token
                refreshToken = refreshed.refreshToken ?: cached.refreshToken,
                expiresAtEpoch = Instant.now().epochSecond + refreshed.expiresInSeconds,
            )
            spotifyTokenCache = newToken
            tokenStore.saveSpotifyToken(newToken)
            newToken.accessToken
        } catch (e: Exception) {
            _spotifyAuthState.value = AuthState.Error(
                message = "Spotify session expired. Please re-authenticate.",
                isRecoverable = true,
            )
            throw AuthError.TokenRefreshFailed(AuthService.SPOTIFY, e)
        }
    }

    private suspend fun getYouTubeAccessToken(): String = youTubeMutex.withLock {
        val cached = youTubeTokenCache
            ?: throw AuthError.InvalidToken(AuthService.YOUTUBE_MUSIC)

        if (!cached.isExpiringSoon) return cached.accessToken

        return try {
            val refreshed = youTubeRefresher.refresh(cached.refreshToken)
            val newToken = cached.copy(
                accessToken = refreshed.accessToken,
                refreshToken = refreshed.refreshToken ?: cached.refreshToken,
                expiresAtEpoch = Instant.now().epochSecond + refreshed.expiresInSeconds,
            )
            youTubeTokenCache = newToken
            tokenStore.saveYouTubeToken(newToken)
            newToken.accessToken
        } catch (e: Exception) {
            _youTubeAuthState.value = AuthState.Error(
                message = "YouTube Music session expired. Please re-authenticate.",
                isRecoverable = true,
            )
            throw AuthError.TokenRefreshFailed(AuthService.YOUTUBE_MUSIC, e)
        }
    }

    override suspend fun onTokensReceived(
        service: AuthService,
        accessToken: String,
        refreshToken: String,
        expiresInSeconds: Long,
        scope: String,
    ) {
        val token = ServiceToken(
            accessToken = accessToken,
            refreshToken = refreshToken,
            expiresAtEpoch = Instant.now().epochSecond + expiresInSeconds,
            scope = scope,
        )
        when (service) {
            AuthService.SPOTIFY -> {
                spotifyTokenCache = token
                tokenStore.saveSpotifyToken(token)
                _spotifyAuthState.value = AuthState.Connecting // user info fetched next
            }
            AuthService.YOUTUBE_MUSIC -> {
                youTubeTokenCache = token
                tokenStore.saveYouTubeToken(token)
                _youTubeAuthState.value = AuthState.Connecting
            }
        }
    }

    /** Called after user profile is fetched post-auth to finalize the Connected state. */
    suspend fun onUserInfoReceived(service: AuthService, user: UserInfo) {
        when (service) {
            AuthService.SPOTIFY -> {
                spotifyTokenCache?.let { tokenStore.saveSpotifyToken(it, user) }
                _spotifyAuthState.value = AuthState.Connected(user)
            }
            AuthService.YOUTUBE_MUSIC -> {
                youTubeTokenCache?.let { tokenStore.saveYouTubeToken(it, user) }
                _youTubeAuthState.value = AuthState.Connected(user)
            }
        }
    }

    override suspend fun disconnect(service: AuthService) {
        when (service) {
            AuthService.SPOTIFY -> {
                spotifyTokenCache = null
                tokenStore.clearSpotify()
                _spotifyAuthState.value = AuthState.NotConnected
            }
            AuthService.YOUTUBE_MUSIC -> {
                youTubeTokenCache = null
                tokenStore.clearYouTube()
                _youTubeAuthState.value = AuthState.NotConnected
            }
        }
    }
}
```

```kotlin
package com.stash.core.auth.spotify

import android.content.Context
import android.content.Intent
import android.net.Uri
import dagger.hilt.android.qualifiers.ApplicationContext
import net.openid.appauth.AuthorizationException
import net.openid.appauth.AuthorizationRequest
import net.openid.appauth.AuthorizationResponse
import net.openid.appauth.AuthorizationService
import net.openid.appauth.AuthorizationServiceConfiguration
import net.openid.appauth.ResponseTypeValues
import net.openid.appauth.TokenResponse
import javax.inject.Inject
import javax.inject.Singleton
import kotlin.coroutines.resume
import kotlin.coroutines.resumeWithException
import kotlin.coroutines.suspendCoroutine

@Singleton
class SpotifyAuthManager @Inject constructor(
    @ApplicationContext private val context: Context,
) {
    private val serviceConfig = AuthorizationServiceConfiguration(
        SpotifyAuthConfig.AUTH_ENDPOINT,
        SpotifyAuthConfig.TOKEN_ENDPOINT,
    )

    private val authService: AuthorizationService by lazy {
        AuthorizationService(context)
    }

    /**
     * Creates the authorization Intent that opens Custom Chrome Tab.
     * The caller (Activity/ViewModel) launches this via ActivityResultLauncher.
     */
    fun createAuthIntent(): Intent {
        val authRequest = AuthorizationRequest.Builder(
            serviceConfig,
            SpotifyAuthConfig.CLIENT_ID,
            ResponseTypeValues.CODE,
            SpotifyAuthConfig.REDIRECT_URI,
        )
            .setScope(SpotifyAuthConfig.SCOPES)
            .setCodeVerifierChallenge() // AppAuth generates PKCE verifier+challenge automatically
            .build()

        return authService.getAuthorizationRequestIntent(authRequest)
    }

    /**
     * Handles the OAuth callback Intent from the redirect.
     * Exchanges the authorization code for tokens.
     *
     * @return Triple of (accessToken, refreshToken, expiresInSeconds)
     */
    suspend fun handleAuthResponse(intent: Intent): SpotifyTokenResult {
        val response = AuthorizationResponse.fromIntent(intent)
        val exception = AuthorizationException.fromIntent(intent)

        if (exception != null) {
            throw exception
        }

        requireNotNull(response) { "No authorization response in intent" }

        return suspendCoroutine { continuation ->
            authService.performTokenRequest(response.createTokenExchangeRequest()) { tokenResponse, ex ->
                if (ex != null) {
                    continuation.resumeWithException(ex)
                    return@performTokenRequest
                }

                requireNotNull(tokenResponse) { "Token response was null" }

                continuation.resume(
                    SpotifyTokenResult(
                        accessToken = tokenResponse.accessToken!!,
                        refreshToken = tokenResponse.refreshToken!!,
                        expiresInSeconds = tokenResponse.accessTokenExpirationTime?.let {
                            (it - System.currentTimeMillis()) / 1000
                        } ?: SpotifyAuthConfig.TOKEN_LIFETIME_SECONDS,
                        scope = tokenResponse.scope ?: SpotifyAuthConfig.SCOPES,
                    )
                )
            }
        }
    }

    fun dispose() {
        authService.dispose()
    }
}

data class SpotifyTokenResult(
    val accessToken: String,
    val refreshToken: String,
    val expiresInSeconds: Long,
    val scope: String,
)
```

```kotlin
import okhttp3.OkHttpClient
import okhttp3.Request

import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class SpotifyTokenRefresher @Inject constructor(
    private val httpClient: OkHttpClient,
    private val json: Json,
) {
    /**
     * Refresh the Spotify access token using the refresh token.
     * Spotify PKCE refresh does NOT require client_secret.
     *
     * POST https://accounts.spotify.com/api/token
     * Content-Type: application/x-www-form-urlencoded
     * Body: grant_type=refresh_token&refresh_token=xxx&client_id=xxx

    suspend fun refresh(refreshToken: String): RefreshResult = withContext(Dispatchers.IO) {
        val body = FormBody.Builder()
            .add("grant_type", "refresh_token")
            .add("refresh_token", refreshToken)
            .add("client_id", SpotifyAuthConfig.CLIENT_ID)
            .build()

        val request = Request.Builder()
            .url("https://accounts.spotify.com/api/token")
            .post(body)
            .build()

        val response = httpClient.newCall(request).execute()

        if (!response.isSuccessful) {
            throw SpotifyRefreshException(
                code = response.code,
                body = response.body?.string() ?: "No body",
            )
        }

        val responseBody = response.body!!.string()
        val tokenResponse = json.decodeFromString<SpotifyRefreshResponse>(responseBody)

        RefreshResult(
            accessToken = tokenResponse.access_token,
            expiresInSeconds = tokenResponse.expires_in,
            // Spotify may rotate the refresh token
            refreshToken = tokenResponse.refresh_token,
        )
    }
}
```

```kotlin
package com.stash.core.auth.youtube

import com.stash.core.auth.model.AuthError
import com.stash.core.auth.model.AuthService
import com.stash.core.auth.model.DeviceCodeState
import kotlinx.coroutines.CancellationException
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.delay
import kotlinx.coroutines.withContext
import kotlinx.serialization.json.Json
import okhttp3.FormBody
import okhttp3.OkHttpClient
import okhttp3.Request
import java.time.Instant
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class YouTubeDeviceFlowManager @Inject constructor(
    private val httpClient: OkHttpClient,
    private val json: Json,
) {
    /**
     * Step 1: Request a device code.
     * User must visit [DeviceCodeState.verificationUrl] and enter [DeviceCodeState.userCode].
     */
    suspend fun requestDeviceCode(): DeviceCodeState = withContext(Dispatchers.IO) {
        val body = FormBody.Builder()
            .add("client_id", YouTubeAuthConfig.CLIENT_ID)
            .add("scope", YouTubeAuthConfig.SCOPE)
            .build()

        val request = Request.Builder()
            .url(YouTubeAuthConfig.DEVICE_CODE_URL)
            .post(body)
            .build()

        val response = httpClient.newCall(request).execute()
        if (!response.isSuccessful) {
            throw Exception("Device code request failed: ${response.code} ${response.body?.string()}")
        }

        val dcResponse = json.decodeFromString<DeviceCodeResponse>(response.body!!.string())

        DeviceCodeState(
            deviceCode = dcResponse.deviceCode,
            userCode = dcResponse.userCode,
            verificationUrl = dcResponse.verificationUrl,
            expiresAtEpoch = Instant.now().epochSecond + dcResponse.expiresIn,
            intervalSeconds = dcResponse.interval,
        )
    }

    /**
     * Step 2: Poll for authorization.
     * This suspends until the user completes authorization, the code expires, or is cancelled.
     *
     * @return Token result on success
     * @throws AuthError.DeviceCodeExpired if the device code expires
     * @throws AuthError.AuthorizationDenied if user denies access
     * @throws CancellationException if the coroutine is cancelled (user navigates away)
     */
    suspend fun pollForAuthorization(deviceCode: DeviceCodeState): YouTubeTokenResult =
        withContext(Dispatchers.IO) {
            var intervalMs = deviceCode.intervalSeconds * 1000L

            while (true) {
                // Check expiration
                if (Instant.now().epochSecond >= deviceCode.expiresAtEpoch) {
                    throw AuthError.DeviceCodeExpired(AuthService.YOUTUBE_MUSIC)
                }

                delay(intervalMs)

                val body = FormBody.Builder()
                    .add("client_id", YouTubeAuthConfig.CLIENT_ID)
                    .add("client_secret", YouTubeAuthConfig.CLIENT_SECRET)
                    .add("device_code", deviceCode.deviceCode)
                    .add("grant_type", YouTubeAuthConfig.DEVICE_GRANT_TYPE)
                    .build()

                val request = Request.Builder()
                    .url(YouTubeAuthConfig.TOKEN_URL)
                    .post(body)
                    .build()

                val response = httpClient.newCall(request).execute()
                val responseBody = response.body!!.string()

                if (response.isSuccessful) {
                    val tokenResponse = json.decodeFromString<GoogleTokenResponse>(responseBody)
                    return@withContext YouTubeTokenResult(
                        accessToken = tokenResponse.accessToken,
                        refreshToken = tokenResponse.refreshToken!!,
                        expiresInSeconds = tokenResponse.expiresIn,
                        scope = tokenResponse.scope ?: YouTubeAuthConfig.SCOPE,
                    )
                }

                // Parse error to determine action
                val error = json.decodeFromString<GoogleTokenError>(responseBody)
                when (error.error) {
                    "authorization_pending" -> continue // Normal -- keep polling
                    "slow_down" -> {
                        intervalMs += 5000L // Back off by 5 seconds as spec requires
                        continue
                    }
                    "access_denied" -> throw AuthError.AuthorizationDenied(AuthService.YOUTUBE_MUSIC)
                    "expired_token" -> throw AuthError.DeviceCodeExpired(AuthService.YOUTUBE_MUSIC)
                    else -> throw Exception("YouTube auth error: ${error.error} - ${error.errorDescription}")
                }
            }

            @Suppress("UNREACHABLE_CODE")
            error("Unreachable")
        }
}

data class YouTubeTokenResult(
    val accessToken: String,
    val refreshToken: String,
    val expiresInSeconds: Long,
    val scope: String,
)
```

#### Task E.5 -- Auth interceptor for OkHttp
**Time:** 4 min | **File:** `core/auth/src/main/java/com/stash/core/auth/interceptor/AuthInterceptor.kt`

```kotlin
package com.stash.core.auth.interceptor

import com.stash.core.auth.TokenManager
import com.stash.core.auth.model.AuthService
import kotlinx.coroutines.runBlocking
import okhttp3.Interceptor
import okhttp3.Response

/**
 * OkHttp interceptor that attaches Bearer tokens to requests.
 *
 * The target service is determined by the request URL:
 * - api.spotify.com -> Spotify token
 * - music.youtube.com -> YouTube Music token
 *
 * The interceptor handles token refresh automatically via TokenManager.
 */
class AuthInterceptor(
    private val tokenManager: TokenManager,
) : Interceptor {

    override fun intercept(chain: Interceptor.Chain): Response {
        val request = chain.request()
        val host = request.url.host

        val service = when {
            host.contains("spotify.com") -> AuthService.SPOTIFY
            host.contains("youtube.com") || host.contains("googleapis.com") -> AuthService.YOUTUBE_MUSIC
            else -> return chain.proceed(request) // No auth needed
        }

        // runBlocking is acceptable in an OkHttp interceptor because
        // OkHttp calls run on its own dispatcher (Dispatchers.IO equivalent)
        val token = runBlocking { tokenManager.getAccessToken(service) }

        val authenticatedRequest = request.newBuilder()
            .header("Authorization", "Bearer $token")
            .build()

        val response = chain.proceed(authenticatedRequest)

        // If 401, try refreshing once and retry
        if (response.code == 401) {
            response.close()

            val refreshedToken = runBlocking { tokenManager.getAccessToken(service) }
            val retryRequest = request.newBuilder()
                .header("Authorization", "Bearer $refreshedToken")
                .build()

            return chain.proceed(retryRequest)
        }

        return response
    }
}
```

```kotlin
import com.stash.data.spotify.model.PlaylistTrack

import kotlinx.coroutines.withContext
import kotlinx.serialization.json.Json
import okhttp3.OkHttpClient
import okhttp3.Request
import javax.inject.Inject
import javax.inject.Named
import javax.inject.Singleton

@Singleton
class SpotifyApiClient @Inject constructor(
    @Named("authenticated") private val httpClient: OkHttpClient,
    private val json: Json,
) {
    companion object {
        private const val BASE_URL = "https://api.spotify.com/v1"
        private const val RATE_LIMIT_DELAY_MS = 100L // 100ms between calls during bulk sync
    }

    /** GET /v1/me -- requires user-read-private scope */
    suspend fun getCurrentUser(): SpotifyUser = withContext(Dispatchers.IO) {
        val request = Request.Builder()
            .url("$BASE_URL/me")
            .get()
            .build()

        executeAndParse(request)
    }

    /**
     * GET /v1/me/playlists -- paginated, 50 per page.
     * Requires playlist-read-private scope.
     */
    suspend fun getCurrentUserPlaylists(
        limit: Int = 50,
        offset: Int = 0,
    ): PagingObject<SimplifiedPlaylist> = withContext(Dispatchers.IO) {
        val request = Request.Builder()
            .url("$BASE_URL/me/playlists?limit=$limit&offset=$offset")
            .get()
            .build()

        executeAndParse(request)
    }

    /** Fetch ALL playlists by paginating. */
    suspend fun getAllPlaylists(): List<SimplifiedPlaylist> {
        val all = mutableListOf<SimplifiedPlaylist>()
        var offset = 0
        val limit = 50

        do {
            val page = getCurrentUserPlaylists(limit = limit, offset = offset)
            all.addAll(page.items)
            offset += limit
            if (page.next != null) delay(RATE_LIMIT_DELAY_MS)
        } while (page.next != null)

        return all
    }

    /**
     * GET /v1/playlists/{id}/tracks -- paginated, 100 per page.
     */
    suspend fun getPlaylistTracks(
        playlistId: String,
        limit: Int = 100,
        offset: Int = 0,
    ): PagingObject<PlaylistTrack> = withContext(Dispatchers.IO) {
        val request = Request.Builder()
            .url("$BASE_URL/playlists/$playlistId/tracks?limit=$limit&offset=$offset")
            .get()
            .build()

        executeAndParse(request)
    }

    private suspend inline fun <reified T> executeAndParse(request: Request): T =
        withContext(Dispatchers.IO) {
            val response = httpClient.newCall(request).execute()

            if (response.code == 429) {
                // Respect Retry-After header
                val retryAfter = response.header("Retry-After")?.toLongOrNull() ?: 5
                response.close()
                delay(retryAfter * 1000)
                // Retry once
                val retryResponse = httpClient.newCall(request).execute()
                if (!retryResponse.isSuccessful) {
                    throw SpotifyApiException(retryResponse.code, retryResponse.body?.string())
                }
                return@withContext json.decodeFromString<T>(retryResponse.body!!.string())
            }

            if (!response.isSuccessful) {
                throw SpotifyApiException(response.code, response.body?.string())
            }

            json.decodeFromString<T>(response.body!!.string())
        }
}
```

```kotlin
/**
 * Filters the user's playlists to find Spotify Daily Mixes (1-6).
 *
 * Daily Mixes are identified by:
 * - owner.id == "spotify"
 * - name matches "Daily Mix 1" through "Daily Mix 6"
 *
 * Users must have opened their Daily Mixes in Spotify at least once
 * for them to appear in their library.
 */
object DailyMixFilter {

    private val DAILY_MIX_REGEX = Regex("""^Daily Mix ([1-6])$""")

    fun filterDailyMixes(playlists: List<SimplifiedPlaylist>): List<DailyMix> {
        return playlists
            .filter { it.owner.id == "spotify" }
            .mapNotNull { playlist ->
                val match = DAILY_MIX_REGEX.matchEntire(playlist.name) ?: return@mapNotNull null
                val mixNumber = match.groupValues[1].toInt()
                DailyMix(
                    mixNumber = mixNumber,
                    playlist = playlist,
                )
            }
            .sortedBy { it.mixNumber }
    }

    /** Filter for Liked Songs -- uses a special endpoint, not in playlists list */
    fun isLikedSongsPlaylist(playlist: SimplifiedPlaylist): Boolean {
        return playlist.name == "Liked Songs" && playlist.owner.id == "spotify"
    }
}
```

```kotlin
package com.stash.data.ytmusic

import com.stash.core.auth.TokenManager
import com.stash.core.auth.model.AuthService
import com.stash.data.ytmusic.model.BrowseResponse
import com.stash.data.ytmusic.model.InnerTubeBody
import com.stash.data.ytmusic.model.InnerTubeDefaults
import com.stash.data.ytmusic.model.SearchResponse
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import kotlinx.serialization.encodeToString
import kotlinx.serialization.json.Json
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.OkHttpClient
import okhttp3.Request
import okhttp3.RequestBody.Companion.toRequestBody
import javax.inject.Inject
import javax.inject.Named
import javax.inject.Singleton

/**
 * Low-level InnerTube HTTP client.
 *
 * All InnerTube endpoints are POST requests to:
 *   https://music.youtube.com/youtubei/v1/{action}?key={API_KEY}
 *
 * Authentication is via Bearer token in Authorization header.
 * The WEB_REMIX client context is used for all requests.
 */
@Singleton
class InnerTubeClient @Inject constructor(
    @Named("base") private val httpClient: OkHttpClient, // NOT authenticated -- we add auth manually
    private val tokenManager: TokenManager,
    private val json: Json,
) {
    private val jsonMediaType = "application/json".toMediaType()

    suspend fun browse(
        browseId: String,
        params: String? = null,
        continuation: String? = null,
    ): BrowseResponse = withContext(Dispatchers.IO) {
        val body = InnerTubeBody(
            context = InnerTubeDefaults.webRemixClient(),
            browseId = if (continuation == null) browseId else null,
            params = params,
            continuation = continuation,
        )

        val endpoint = if (continuation != null) "browse" else "browse"
        executePost<BrowseResponse>(endpoint, body)
    }

    suspend fun search(
        query: String,
        params: String? = null,
    ): SearchResponse = withContext(Dispatchers.IO) {
        val body = InnerTubeBody(
            context = InnerTubeDefaults.webRemixClient(),
            query = query,
            params = params,
        )

        executePost<SearchResponse>("search", body)
    }

    private suspend inline fun <reified T> executePost(
        action: String,
        body: InnerTubeBody,
    ): T {
        val accessToken = try {
            tokenManager.getAccessToken(AuthService.YOUTUBE_MUSIC)
        } catch (e: Exception) {
            null // Allow unauthenticated requests for some endpoints
        }

        val jsonBody = json.encodeToString(body)

        val requestBuilder = Request.Builder()
            .url("${InnerTubeDefaults.BASE_URL}/$action?key=${InnerTubeDefaults.API_KEY}")
            .post(jsonBody.toRequestBody(jsonMediaType))
            .header("User-Agent", InnerTubeDefaults.USER_AGENT)
            .header("Referer", InnerTubeDefaults.REFERER)
            .header("Content-Type", "application/json")
            .header("X-Goog-Api-Format-Version", "1")
            .header("X-YouTube-Client-Name", "67") // WEB_REMIX numeric ID
            .header("X-YouTube-Client-Version", "1.20240101.01.00")
            .header("Origin", "https://music.youtube.com")

        if (accessToken != null) {
            requestBuilder.header("Authorization", "Bearer $accessToken")
        }

        val response = httpClient.newCall(requestBuilder.build()).execute()

        if (!response.isSuccessful) {
            throw InnerTubeException(response.code, response.body?.string())
        }

        return json.decodeFromString<T>(response.body!!.string())
    }
}

class InnerTubeException(val code: Int, body: String?) :
    Exception("InnerTube API error ($code): ${body?.take(500)}")
```

```kotlin
/**
 * Fetches the authenticated user's YouTube channel info
 * for display in the Settings account connection UI.
 *
 * Uses YouTube Data API v3 (not InnerTube) since it has stable endpoints.
 */
@Singleton
class YTMusicAccountService @Inject constructor(
    @Named("authenticated") private val httpClient: OkHttpClient,
    private val json: Json,
) {
    suspend fun getCurrentUser(): UserInfo = withContext(Dispatchers.IO) {
        val request = Request.Builder()
            .url("https://www.googleapis.com/youtube/v3/channels?part=snippet&mine=true")
            .get()
            .build()

        val response = httpClient.newCall(request).execute()
        if (!response.isSuccessful) {
            throw Exception("Failed to fetch YouTube channel info: ${response.code}")
        }

        val body = json.decodeFromString<YouTubeChannelResponse>(response.body!!.string())
        val channel = body.items.firstOrNull()
            ?: throw Exception("No YouTube channel found for authenticated user")

        UserInfo(
            id = channel.id,
            displayName = channel.snippet.title,
            imageUrl = channel.snippet.thumbnails?.default?.url,
        )
    }
}
```

```kotlin
import com.stash.core.auth.model.AuthError
import com.stash.core.auth.model.AuthService

/**
 * Centralized auth error classification and messaging.
 * Maps HTTP errors and exceptions to user-friendly messages.
 */
object AuthErrorHandler {

    fun classifyHttpError(service: AuthService, code: Int, body: String?): AuthError {
        return when (code) {
            401 -> AuthError.InvalidToken(service)
            403 -> AuthError.AuthorizationDenied(service)
            429 -> AuthError.NetworkError(Exception("Rate limited. Please try again later."))
            in 500..599 -> AuthError.NetworkError(
                Exception("${service.name} servers are temporarily unavailable.")
            )
            else -> AuthError.NetworkError(
                Exception("Unexpected error ($code): ${body?.take(200)}")
            )
        }
    }

    fun userMessage(error: AuthError): String {
        return when (error) {
            is AuthError.TokenRefreshFailed -> when (error.service) {
                AuthService.SPOTIFY -> "Your Spotify session expired. Please reconnect in Settings."
                AuthService.YOUTUBE_MUSIC -> "Your YouTube Music session expired. Please reconnect in Settings."
            }
            is AuthError.AuthorizationDenied -> "Authorization was denied. Please try again."
            is AuthError.DeviceCodeExpired -> "The authorization code expired. Please start the process again."
            is AuthError.NetworkError -> error.cause?.message ?: "A network error occurred."
            is AuthError.InvalidToken -> "Your session is invalid. Please reconnect in Settings."
        }
    }
}
```

#### Task I.2 -- Retry interceptor for transient failures
**Time:** 3 min | **File:** `core/network/src/main/java/com/stash/core/network/RetryInterceptor.kt`

```kotlin
package com.stash.core.network

import okhttp3.Interceptor
import okhttp3.Response
import java.io.IOException

/**
 * Retries failed HTTP requests with exponential backoff.
 * Only retries on IOException (network failures) and 5xx server errors.
 * Does NOT retry on 4xx (client errors including auth failures).
 */
class RetryInterceptor(
    private val maxRetries: Int = 3,
    private val initialDelayMs: Long = 1000,
) : Interceptor {

    override fun intercept(chain: Interceptor.Chain): Response {
        val request = chain.request()
        var lastException: IOException? = null

        for (attempt in 0..maxRetries) {
            try {
                val response = chain.proceed(request)

                // Don't retry client errors (4xx)
                if (response.code < 500) return response

                // Retry on server errors
                if (attempt < maxRetries) {
                    response.close()
                    val delay = initialDelayMs * (1L shl attempt) // exponential backoff
                    Thread.sleep(delay.coerceAtMost(30_000))
                    continue
                }

                return response
            } catch (e: IOException) {
                lastException = e
                if (attempt < maxRetries) {
                    val delay = initialDelayMs * (1L shl attempt)
                    Thread.sleep(delay.coerceAtMost(30_000))
                }
            }
        }

        throw lastException ?: IOException("Request failed after $maxRetries retries")
    }
}
```

/**
 * Banner shown at the top of any screen when a service needs re-authentication.
 * Tapping navigates to Settings.
 */
@Composable
fun ReAuthBanner(
    visible: Boolean,
    message: String,
    onNavigateToSettings: () -> Unit,
    modifier: Modifier = Modifier,
) {
    AnimatedVisibility(
        visible = visible,
        enter = slideInVertically { -it },
        exit = slideOutVertically { -it },
        modifier = modifier,
    ) {
        Row(
            verticalAlignment = Alignment.CenterVertically,
            modifier = Modifier
                .fillMaxWidth()
                .clip(RoundedCornerShape(bottomStart = 12.dp, bottomEnd = 12.dp))
                .background(Color(0xFFDC2626).copy(alpha = 0.9f))
                .clickable { onNavigateToSettings() }
                .padding(horizontal = 16.dp, vertical = 12.dp),
        ) {
            Text(
                text = message,
                style = MaterialTheme.typography.bodySmall,
                fontWeight = FontWeight.Medium,
                color = Color.White,
                modifier = Modifier.weight(1f),
            )
            Text(
                text = "Fix",
                style = MaterialTheme.typography.labelMedium,
                fontWeight = FontWeight.Bold,
                color = Color.White,
            )
        }
    }
}

## Key Technical Risks & Mitigations

| Risk | Mitigation |
|------|------------|
| AppAuth `setCodeVerifierChallenge()` API may differ by version | Fallback to explicit `setCodeVerifier(verifier, challenge, S256)` documented in Task D.2 |
| InnerTube response models break on YouTube changes | All models use `ignoreUnknownKeys = true`, nullable fields, and empty list defaults |
| Tink keyset corruption on OS update | `EncryptedAuthSerializer.readFrom()` catches decryption errors and returns default (forces re-auth) |
| Google may restrict YouTube scope on device flow | The `youtube` scope is documented as supported for TV/limited-input devices |
| Spotify may rotate refresh tokens | `SpotifyTokenRefresher` checks for new refresh token in response and persists it |
| OkHttp `runBlocking` in AuthInterceptor | Acceptable since OkHttp interceptors run on IO dispatcher. Alternative: use coroutine-based interceptor if OkHttp 5 is adopted |
