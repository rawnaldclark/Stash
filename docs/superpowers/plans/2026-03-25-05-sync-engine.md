```kotlin
package com.stash.core.data.sync

import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class SyncStateManager @Inject constructor() {

    private val _state = MutableStateFlow<SyncState>(SyncState.Idle)
    val state: StateFlow<SyncState> = _state.asStateFlow()

    private val _lastSyncResult = MutableStateFlow<SyncState.Completed?>(null)
    val lastSyncResult: StateFlow<SyncState.Completed?> = _lastSyncResult.asStateFlow()

    val isSyncing: Boolean
        get() = _state.value !is SyncState.Idle &&
                _state.value !is SyncState.Completed &&
                _state.value !is SyncState.Error

    fun updateState(newState: SyncState) {
        _state.value = newState
        if (newState is SyncState.Completed) {
            _lastSyncResult.value = newState
        }
    }

    fun reset() {
        _state.value = SyncState.Idle
    }

    /**
     * Calculate overall progress [0.0, 1.0] based on phase weights from spec:
     * Auth=5%, Fetch=15%, Diff=5%, Download=70%, Finalize=5%
     */
    fun overallProgress(): Float = when (val s = _state.value) {
        is SyncState.Idle -> 0f
        is SyncState.Authenticating -> s.progress * 0.05f
        is SyncState.FetchingPlaylists -> 0.05f + s.progress * 0.15f
        is SyncState.Diffing -> 0.20f + s.progress * 0.05f
        is SyncState.Downloading -> 0.25f + s.progress * 0.70f
        is SyncState.Finalizing -> 0.95f + s.progress * 0.05f
        is SyncState.Completed -> 1.0f
        is SyncState.Error -> overallProgressForPhase(s.phase)
    }

    private fun overallProgressForPhase(phase: String): Float = when (phase) {
        "auth" -> 0.025f
        "fetch" -> 0.125f
        "diff" -> 0.225f
        "download" -> 0.60f
        "finalize" -> 0.975f
        else -> 0f
    }
}
```

## Task 4: Create TrackMatcher -- canonical normalization and duplicate detection

**File:** `core/data/src/main/kotlin/com/stash/core/data/sync/TrackMatcher.kt`

**Why:** This is the cross-platform deduplication logic. When the same song appears in both Spotify and YouTube Music, we must not download it twice.

```kotlin
package com.stash.core.data.sync

import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class TrackMatcher @Inject constructor() {

    /**
     * Normalize a title for canonical comparison:
     * - Lowercase
     * - Strip content in parentheses and brackets (feat., remix, etc. unless query contains them)
     * - Remove "feat.", "ft.", "featuring"
     * - Collapse whitespace, trim
     */
    fun canonicalTitle(title: String): String {
        return title
            .lowercase()
            .replace(Regex("\\(.*?\\)"), "")
            .replace(Regex("\\[.*?]"), "")
            .replace(Regex("\\b(feat\\.?|ft\\.?|featuring)\\b"), "")
            .replace(Regex("[^a-z0-9\\s]"), "")
            .replace(Regex("\\s+"), " ")
            .trim()
    }

    /**
     * Normalize artist for canonical comparison:
     * - Lowercase
     * - Split on ", " / " & " / " and " / " x " / " vs "
     * - Sort alphabetically
     * - Rejoin with ", "
     */
    fun canonicalArtist(artist: String): String {
        return artist
            .lowercase()
            .split(Regex("[,&x]|\\band\\b|\\bvs\\.?\\b"))
            .map { it.trim() }
            .filter { it.isNotEmpty() }
            .sorted()
            .joinToString(", ")
    }

    /**
     * Check if two tracks are duplicates using exact canonical match
     */
    fun isExactMatch(
        title1: String, artist1: String,
        title2: String, artist2: String,
    ): Boolean {
        return canonicalTitle(title1) == canonicalTitle(title2) &&
               canonicalArtist(artist1) == canonicalArtist(artist2)
    }

    /**
     * Jaro-Winkler similarity [0.0, 1.0]
     */
    fun jaroWinklerSimilarity(s1: String, s2: String): Double {
        if (s1 == s2) return 1.0
        if (s1.isEmpty() || s2.isEmpty()) return 0.0

        val matchDistance = (maxOf(s1.length, s2.length) / 2) - 1
        val s1Matches = BooleanArray(s1.length)
        val s2Matches = BooleanArray(s2.length)
        var matches = 0
        var transpositions = 0

        for (i in s1.indices) {
            val start = maxOf(0, i - matchDistance)
            val end = minOf(i + matchDistance + 1, s2.length)
            for (j in start until end) {
                if (s2Matches[j] || s1[i] != s2[j]) continue
                s1Matches[i] = true
                s2Matches[j] = true
                matches++
                break
            }
        }
        if (matches == 0) return 0.0

        var k = 0
        for (i in s1.indices) {
            if (!s1Matches[i]) continue
            while (!s2Matches[k]) k++
            if (s1[i] != s2[k]) transpositions++
            k++
        }

        val jaro = (matches.toDouble() / s1.length +
                     matches.toDouble() / s2.length +
                     (matches - transpositions / 2.0) / matches) / 3.0

        // Winkler modification: boost for common prefix up to 4 chars
        var prefix = 0
        for (i in 0 until minOf(4, minOf(s1.length, s2.length))) {
            if (s1[i] == s2[i]) prefix++ else break
        }
        return jaro + prefix * 0.1 * (1.0 - jaro)
    }

    /**
     * Fuzzy match: title JW > 0.92 AND artist JW > 0.90, with duration confirmation (<=10s diff)
     */
    fun isFuzzyMatch(
        title1: String, artist1: String, durationMs1: Long,
        title2: String, artist2: String, durationMs2: Long,
    ): Boolean {
        val titleSim = jaroWinklerSimilarity(canonicalTitle(title1), canonicalTitle(title2))
        val artistSim = jaroWinklerSimilarity(canonicalArtist(artist1), canonicalArtist(artist2))
        val durationDiff = kotlin.math.abs(durationMs1 - durationMs2)
        return titleSim > 0.92 && artistSim > 0.90 && durationDiff <= 10_000L
    }
}
```

## Task 6: Create SyncNotificationManager

**File:** `core/data/src/main/kotlin/com/stash/core/data/sync/SyncNotificationManager.kt`

**Why:** Workers need a centralized way to create and update notifications. Two channels: ongoing sync progress (LOW importance, foreground service) and sync summary (DEFAULT importance, posted on completion).

```kotlin
package com.stash.core.data.sync

import android.app.Notification
import android.app.NotificationChannel
import android.app.NotificationManager
import android.app.PendingIntent
import android.content.Context
import android.content.Intent
import android.os.Build
import androidx.core.app.NotificationCompat
import dagger.hilt.android.qualifiers.ApplicationContext
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class SyncNotificationManager @Inject constructor(
    @ApplicationContext private val context: Context,
) {

    companion object {
        const val CHANNEL_SYNC_PROGRESS = "sync_progress"
        const val CHANNEL_SYNC_SUMMARY = "sync_summary"
        const val NOTIFICATION_ID_SYNC_PROGRESS = 1001
        const val NOTIFICATION_ID_SYNC_SUMMARY = 1002
    }

    private val notificationManager =
        context.getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager

    fun createChannels() {
        val progressChannel = NotificationChannel(
            CHANNEL_SYNC_PROGRESS,
            "Sync Progress",
            NotificationManager.IMPORTANCE_LOW,
        ).apply {
            description = "Shows progress while syncing playlists"
            setShowBadge(false)
        }

        val summaryChannel = NotificationChannel(
            CHANNEL_SYNC_SUMMARY,
            "Sync Summary",
            NotificationManager.IMPORTANCE_DEFAULT,
        ).apply {
            description = "Summary after sync completes"
        }

        notificationManager.createNotificationChannels(listOf(progressChannel, summaryChannel))
    }

    fun buildProgressNotification(
        title: String,
        text: String,
        progress: Int, // 0-100
        indeterminate: Boolean = false,
    ): Notification {
        val launchIntent = context.packageManager
            .getLaunchIntentForPackage(context.packageName)
            ?.apply { flags = Intent.FLAG_ACTIVITY_SINGLE_TOP }
        val pendingIntent = PendingIntent.getActivity(
            context, 0, launchIntent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE,
        )

        return NotificationCompat.Builder(context, CHANNEL_SYNC_PROGRESS)
            .setSmallIcon(android.R.drawable.stat_notify_sync) // TODO: Replace with R.drawable.ic_sync
            .setContentTitle(title)
            .setContentText(text)
            .setProgress(100, progress, indeterminate)
            .setOngoing(true)
            .setOnlyAlertOnce(true)
            .setContentIntent(pendingIntent)
            .setForegroundServiceBehavior(NotificationCompat.FOREGROUND_SERVICE_IMMEDIATE)
            .build()
    }

    fun updateProgress(title: String, text: String, progress: Int) {
        notificationManager.notify(
            NOTIFICATION_ID_SYNC_PROGRESS,
            buildProgressNotification(title, text, progress),
        )
    }

    fun showSummary(
        newTracks: Int,
        playlistsChecked: Int,
        failed: Int,
    ) {
        val text = buildString {
            append("Added $newTracks new track${if (newTracks != 1) "s" else ""}")
            append(" from $playlistsChecked playlist${if (playlistsChecked != 1) "s" else ""}")
            if (failed > 0) append(" ($failed failed)")
        }

        val notification = NotificationCompat.Builder(context, CHANNEL_SYNC_SUMMARY)
            .setSmallIcon(android.R.drawable.stat_notify_sync) // TODO: Replace with R.drawable.ic_sync_done
            .setContentTitle("Sync Complete")
            .setContentText(text)
            .setAutoCancel(true)
            .build()

        notificationManager.notify(NOTIFICATION_ID_SYNC_SUMMARY, notification)
    }

    fun cancelProgress() {
        notificationManager.cancel(NOTIFICATION_ID_SYNC_PROGRESS)
    }
}
```

## Task 7: Create PlaylistFetchWorker

**File:** `core/data/src/main/kotlin/com/stash/core/data/sync/workers/PlaylistFetchWorker.kt`

**Why:** First worker in the chain. Fetches all playlists + tracks from both Spotify and YouTube Music APIs, writes them into the `remote_playlist_snapshot` / `remote_track_snapshot` tables for DiffWorker.

```kotlin
package com.stash.core.data.sync.workers

import android.content.Context
import androidx.hilt.work.HiltWorker
import androidx.work.CoroutineWorker
import androidx.work.WorkerParameters
import androidx.work.workDataOf
import com.stash.core.auth.TokenManager
import com.stash.core.data.db.dao.RemoteSnapshotDao
import com.stash.core.data.db.dao.SyncHistoryDao
import com.stash.core.data.db.entity.RemotePlaylistSnapshotEntity
import com.stash.core.data.db.entity.RemoteTrackSnapshotEntity
import com.stash.core.data.db.entity.SyncHistoryEntity
import com.stash.core.data.sync.SyncState
import com.stash.core.data.sync.SyncStateManager
import com.stash.data.spotify.SpotifyApiClient
import com.stash.data.ytmusic.InnerTubeClient
import dagger.assisted.Assisted
import dagger.assisted.AssistedInject
import kotlinx.coroutines.delay

@HiltWorker
class PlaylistFetchWorker @AssistedInject constructor(
    @Assisted appContext: Context,
    @Assisted workerParams: WorkerParameters,
    private val spotifyApi: SpotifyApiClient,
    private val innerTubeClient: InnerTubeClient,
    private val tokenManager: TokenManager,
    private val snapshotDao: RemoteSnapshotDao,
    private val syncHistoryDao: SyncHistoryDao,
    private val syncStateManager: SyncStateManager,
) : CoroutineWorker(appContext, workerParams) {

    companion object {
        const val KEY_SYNC_ID = "sync_id"
        const val KEY_TRIGGER = "trigger" // "scheduled" or "manual"
    }

    override suspend fun doWork(): Result {
        // Create sync history record
        val trigger = inputData.getString(KEY_TRIGGER) ?: "scheduled"
        val syncId = syncHistoryDao.insert(
            SyncHistoryEntity(
                startedAt = System.currentTimeMillis(),
                status = "IN_PROGRESS",
                trigger = trigger,
            )
        )

        try {
            // Phase: Authenticating
            syncStateManager.updateState(SyncState.Authenticating(0f))
            tokenManager.ensureValidTokens() // Refresh if needed
            syncStateManager.updateState(SyncState.Authenticating(1f))

            // Phase: Fetching Playlists
            syncStateManager.updateState(SyncState.FetchingPlaylists(0f, "Spotify"))

            // Clean old snapshots for this sync
            snapshotDao.deletePlaylistSnapshots(syncId)
            snapshotDao.deleteTrackSnapshots(syncId)

            var playlistsFetched = 0
            val totalEstimated = 10 // Rough estimate: 6 daily mixes + liked + 3 YT playlists

            // --- Spotify ---
            if (tokenManager.hasSpotifyToken()) {
                val playlists = spotifyApi.getUserPlaylists()

                // Filter Daily Mixes (owned by "spotify", name matches "Daily Mix N")
                val dailyMixes = playlists.filter { p ->
                    p.owner.id == "spotify" && p.name.matches(Regex("Daily Mix [1-6]"))
                }

                // Liked Songs (special endpoint)
                val allSpotifyPlaylists = dailyMixes + playlists.filter { p ->
                    // Include any user playlists that are active in Stash
                    p.owner.id != "spotify" // User's own playlists
                }

                for (playlist in dailyMixes) {
                    val snapshotEntity = RemotePlaylistSnapshotEntity(
                        syncId = syncId,
                        source = "SPOTIFY",
                        sourcePlaylistId = playlist.id,
                        playlistName = playlist.name,
                        playlistType = "DAILY_MIX",
                        mixNumber = playlist.name.substringAfter("Daily Mix ").toIntOrNull(),
                        snapshotId = playlist.snapshotId,
                        trackCount = playlist.tracks.total,
                        artUrl = playlist.images.firstOrNull()?.url,
                    )
                    val playlistSnapshotId = snapshotDao.insertPlaylistSnapshot(snapshotEntity)

                    // Fetch tracks for this playlist
                    val tracks = spotifyApi.getPlaylistTracks(playlist.id)
                    val trackSnapshots = tracks.mapIndexed { index, track ->
                        RemoteTrackSnapshotEntity(
                            syncId = syncId,
                            snapshotPlaylistId = playlistSnapshotId,
                            title = track.name,
                            artist = track.artists.joinToString(", ") { it.name },
                            album = track.album?.name,
                            durationMs = track.durationMs,
                            spotifyUri = track.uri,
                            albumArtUrl = track.album?.images?.firstOrNull()?.url,
                            position = index,
                        )
                    }
                    snapshotDao.insertTrackSnapshots(trackSnapshots)
                    playlistsFetched++
                    delay(100) // Rate limit: 100ms between Spotify calls per spec
                    syncStateManager.updateState(
                        SyncState.FetchingPlaylists(
```

## Task 8: Create DiffWorker

**File:** `core/data/src/main/kotlin/com/stash/core/data/sync/workers/DiffWorker.kt`

**Why:** Compares remote snapshots against local Room DB. Uses Spotify `snapshot_id` for incremental skip. Produces entries in `download_queue` for new tracks. Uses `TrackMatcher` for cross-platform dedup.

```kotlin
package com.stash.core.data.sync.workers

import android.content.Context
import androidx.hilt.work.HiltWorker
import androidx.work.CoroutineWorker
import androidx.work.WorkerParameters
import androidx.work.workDataOf
import com.stash.core.data.db.dao.DownloadQueueDao
import com.stash.core.data.db.dao.PlaylistDao
import com.stash.core.data.db.dao.PlaylistTrackCrossRefDao
import com.stash.core.data.db.dao.RemoteSnapshotDao
import com.stash.core.data.db.dao.TrackDao
import com.stash.core.data.db.entity.DownloadQueueEntity
import com.stash.core.data.db.entity.PlaylistEntity
import com.stash.core.data.db.entity.PlaylistTrackCrossRef
import com.stash.core.data.db.entity.TrackEntity
import com.stash.core.data.sync.SyncState
import com.stash.core.data.sync.SyncStateManager
import com.stash.core.data.sync.TrackMatcher
import dagger.assisted.Assisted
import dagger.assisted.AssistedInject

@HiltWorker
class DiffWorker @AssistedInject constructor(
    @Assisted appContext: Context,
    @Assisted workerParams: WorkerParameters,
    private val snapshotDao: RemoteSnapshotDao,
    private val playlistDao: PlaylistDao,
    private val trackDao: TrackDao,
    private val crossRefDao: PlaylistTrackCrossRefDao,
    private val downloadQueueDao: DownloadQueueDao,
    private val trackMatcher: TrackMatcher,
    private val syncStateManager: SyncStateManager,
) : CoroutineWorker(appContext, workerParams) {

    override suspend fun doWork(): Result {
        val syncId = inputData.getLong(PlaylistFetchWorker.KEY_SYNC_ID, -1L)
        if (syncId == -1L) return Result.failure()

        try {
            syncStateManager.updateState(SyncState.Diffing(0f))

            val playlistSnapshots = snapshotDao.getPlaylistSnapshots(syncId)
            var newTracksFound = 0
            var playlistsProcessed = 0

            for (playlistSnapshot in playlistSnapshots) {
                // --- Spotify snapshot_id optimization ---
                if (playlistSnapshot.source == "SPOTIFY" && playlistSnapshot.snapshotId != null) {
                    val localSnapshotId = playlistDao.getSnapshotId(
                        playlistSnapshot.sourcePlaylistId,
                        "SPOTIFY",
                    )
                    if (localSnapshotId == playlistSnapshot.snapshotId) {
                        // Playlist unchanged, skip entirely
                        playlistsProcessed++
                        syncStateManager.updateState(
                            SyncState.Diffing(
                                progress = playlistsProcessed.toFloat() / playlistSnapshots.size,
                                playlistsProcessed = playlistsProcessed,
                                playlistsTotal = playlistSnapshots.size,
                                newTracksFound = newTracksFound,
                            )
                        )
                        continue
                    }
                }

                // Ensure local playlist exists
                val localPlaylist = playlistDao.getBySourceId(
                    playlistSnapshot.sourcePlaylistId,
                    playlistSnapshot.source,
                ) ?: run {
                    // Create local playlist
                    val entity = PlaylistEntity(
                        name = playlistSnapshot.playlistName,
                        source = playlistSnapshot.source,
                        sourceId = playlistSnapshot.sourcePlaylistId,
                        type = playlistSnapshot.playlistType,
                        mixNumber = playlistSnapshot.mixNumber,
                        trackCount = playlistSnapshot.trackCount,
                        isActive = true,
                        artUrl = playlistSnapshot.artUrl,
                        snapshotId = playlistSnapshot.snapshotId,
                    )
                    val id = playlistDao.insert(entity)
                    entity.copy(id = id)
                }

                // Get remote tracks for this playlist
                val remoteTracks = snapshotDao.getTrackSnapshots(playlistSnapshot.id)

                // Get all local tracks for cross-platform dedup
                // We load them once per diff cycle for efficiency
                val localTracksBySpotifyUri = trackDao.getAllSpotifyUriMap()
                val localTracksByYoutubeId = trackDao.getAllYoutubeIdMap()

                val currentCrossRefTrackIds = mutableSetOf<Long>()

                for (remoteTrack in remoteTracks) {
                    // Check 1: Exact ID match (Spotify URI or YouTube ID)
                    var existingTrack: TrackEntity? = null

                    if (remoteTrack.spotifyUri != null) {
                        existingTrack = localTracksBySpotifyUri[remoteTrack.spotifyUri]
                    }
                    if (existingTrack == null && remoteTrack.youtubeId != null) {
                        existingTrack = localTracksByYoutubeId[remoteTrack.youtubeId]
                    }

                    // Check 2: Canonical match (cross-platform dedup)
                    if (existingTrack == null) {
                        val canonical = trackMatcher.canonicalTitle(remoteTrack.title) to
                                        trackMatcher.canonicalArtist(remoteTrack.artist)
                        existingTrack = trackDao.findByCanonical(canonical.first, canonical.second)
                    }

                    if (existingTrack != null) {
                        // Track exists locally -- update cross-source linking if needed
                        if (remoteTrack.spotifyUri != null && existingTrack.spotifyUri == null) {
                            trackDao.updateSpotifyUri(existingTrack.id, remoteTrack.spotifyUri)
                            trackDao.updateSource(existingTrack.id, "BOTH")
                        }
                        if (remoteTrack.youtubeId != null && existingTrack.youtubeId == null) {
                            trackDao.updateYoutubeId(existingTrack.id, remoteTrack.youtubeId)
                            trackDao.updateSource(existingTrack.id, "BOTH")
                        }
                        currentCrossRefTrackIds.add(existingTrack.id)
                    } else {
                        // New track -- insert and enqueue for download
                        val trackEntity = TrackEntity(
                            title = remoteTrack.title,
                            artist = remoteTrack.artist,
                            album = remoteTrack.album,
                            durationMs = remoteTrack.durationMs,
                            source = playlistSnapshot.source,
                            spotifyUri = remoteTrack.spotifyUri,
                            youtubeId = remoteTrack.youtubeId,
                            albumArtUrl = remoteTrack.albumArtUrl,
                            canonicalTitle = trackMatcher.canonicalTitle(remoteTrack.title),
                            canonicalArtist = trackMatcher.canonicalArtist(remoteTrack.artist),
                            isDownloaded = false,
                            dateAdded = System.currentTimeMillis(),
                        )
                        val trackId = trackDao.insert(trackEntity)
                        currentCrossRefTrackIds.add(trackId)

                        // Build search query for yt-dlp
                        val searchQuery = if (remoteTrack.youtubeId != null) {
                            // YouTube track -- we have the direct ID
                            "https://music.youtube.com/watch?v=${remoteTrack.youtubeId}"
                        } else {
                            // Spotify track -- needs yt-dlp search
                            "${remoteTrack.artist} - ${remoteTrack.title}"
                        }

                        downloadQueueDao.insert(
                            DownloadQueueEntity(
                                trackId = trackId,
                                syncId = syncId,
                                status = "QUEUED",
                                searchQuery = searchQuery,
                                youtubeUrl = remoteTrack.youtubeId?.let {
                                    "https://music.youtube.com/watch?v=$it"
                                },
                                createdAt = System.currentTimeMillis(),
                            )
                        )
                        newTracksFound++
                    }
                }

                // Update playlist-track cross refs
                // Add new associations
                for (trackId in currentCrossRefTrackIds) {
                    val position = remoteTracks.indexOfFirst { rt ->
                        // Match back to position
                        true // simplified -- use actual matching
                    }
                    crossRefDao.insertOrIgnore(
                        PlaylistTrackCrossRef(
                            playlistId = localPlaylist.id,
                            trackId = trackId,
                            position = position,
                            addedAt = System.currentTimeMillis(),
                        )
                    )
                }

                // Soft-delete tracks removed from remote playlist
                val existingCrossRefs = crossRefDao.getTrackIdsForPlaylist(localPlaylist.id)
                val removedTrackIds = existingCrossRefs - currentCrossRefTrackIds
                for (removedId in removedTrackIds) {
                    crossRefDao.softDelete(localPlaylist.id, removedId, System.currentTimeMillis())
                }

                // Update snapshot_id for Spotify playlists
                if (playlistSnapshot.source == "SPOTIFY" && playlistSnapshot.snapshotId != null) {
                    playlistDao.updateSnapshotId(
                        playlistSnapshot.sourcePlaylistId,
                        "SPOTIFY",
                        playlistSnapshot.snapshotId,
                    )
                }

                playlistsProcessed++
                syncStateManager.updateState(
                    SyncState.Diffing(
                        progress = playlistsProcessed.toFloat() / playlistSnapshots.size,
                        playlistsProcessed = playlistsProcessed,
                        playlistsTotal = playlistSnapshots.size,
                        newTracksFound = newTracksFound,
                    )
                )
            }

            syncStateManager.updateState(
                SyncState.Diffing(1f, playlistsProcessed, playlistSnapshots.size, newTracksFound)
            )

            return Result.success(
                workDataOf(
                    PlaylistFetchWorker.KEY_SYNC_ID to syncId,
                    "new_tracks_found" to newTracksFound,
                    "playlists_checked" to playlistsProcessed,
                )
            )
        } catch (e: Exception) {
            syncStateManager.updateState(
                SyncState.Error(
                    message = e.message ?: "Diff failed",
                    phase = "diff",
                    isRetryable = true,
                )
            )
            return if (runAttemptCount < 2) Result.retry() else Result.failure(
                workDataOf(PlaylistFetchWorker.KEY_SYNC_ID to syncId)
            )
        }
    }
}
```

## Task 9: Create TrackDownloadWorker

**File:** `core/data/src/main/kotlin/com/stash/core/data/sync/workers/TrackDownloadWorker.kt`

**Why:** The heaviest worker (70% of sync time). Runs as a foreground service. Processes the download queue created by DiffWorker, delegates to the existing DownloadManager from Phase 4.

```kotlin
package com.stash.core.data.sync.workers

import android.content.Context
import android.content.pm.ServiceInfo
import android.os.Build
import androidx.hilt.work.HiltWorker
import androidx.work.CoroutineWorker
import androidx.work.ForegroundInfo
import androidx.work.WorkerParameters
import androidx.work.workDataOf
import com.stash.core.data.db.dao.DownloadQueueDao
import com.stash.core.data.db.dao.SyncHistoryDao
import com.stash.core.data.db.dao.TrackDao
import com.stash.core.data.sync.SyncNotificationManager
import com.stash.core.data.sync.SyncState
import com.stash.core.data.sync.SyncStateManager
import com.stash.data.download.DownloadManager
import dagger.assisted.Assisted
import dagger.assisted.AssistedInject
import kotlinx.coroutines.async
import kotlinx.coroutines.awaitAll
import kotlinx.coroutines.coroutineScope
import kotlinx.coroutines.sync.Semaphore
import kotlinx.coroutines.sync.withPermit

@HiltWorker
class TrackDownloadWorker @AssistedInject constructor(
    @Assisted appContext: Context,
    @Assisted workerParams: WorkerParameters,
    private val downloadQueueDao: DownloadQueueDao,
    private val trackDao: TrackDao,
    private val syncHistoryDao: SyncHistoryDao,
    private val downloadManager: DownloadManager,
    private val syncStateManager: SyncStateManager,
    private val notificationManager: SyncNotificationManager,
) : CoroutineWorker(appContext, workerParams) {

    private val concurrencySemaphore = Semaphore(3) // 3 concurrent downloads per spec

    override suspend fun doWork(): Result {
        val syncId = inputData.getLong(PlaylistFetchWorker.KEY_SYNC_ID, -1L)
        if (syncId == -1L) return Result.failure()

        // Promote to foreground service immediately
        setForeground(createForegroundInfo("Starting downloads...", 0))

        try {
            val queuedItems = downloadQueueDao.getQueuedBySyncId(syncId)
            if (queuedItems.isEmpty()) {
                return Result.success(
                    workDataOf(
                        PlaylistFetchWorker.KEY_SYNC_ID to syncId,
                        "tracks_downloaded" to 0,
                        "tracks_failed" to 0,
                        "bytes_downloaded" to 0L,
                    )
                )
            }

            var tracksCompleted = 0
            var tracksFailed = 0
            var bytesDownloaded = 0L
            val total = queuedItems.size

            syncStateManager.updateState(
                SyncState.Downloading(0f, "", "", 0, total, 0L)
            )

            // Process downloads with concurrency limit of 3
            coroutineScope {
                val results = queuedItems.map { item ->
                    async {
                        concurrencySemaphore.withPermit {
                            if (isStopped) return@withPermit false

                            val track = trackDao.getById(item.trackId) ?: return@withPermit false

                            // Update notification per-track (not per-byte to avoid churn)
                            val currentProgress = ((tracksCompleted.toFloat()) / total * 100).toInt()
                            setForeground(
                                createForegroundInfo(
                                    "${track.artist} - ${track.title}",
                                    currentProgress,
                                )
                            )
                            syncStateManager.updateState(
                                SyncState.Downloading(
                                    progress = tracksCompleted.toFloat() / total,
                                    currentTrackTitle = track.title,
                                    currentTrackArtist = track.artist,
                                    tracksCompleted = tracksCompleted,
                                    tracksTotal = total,
                                    bytesDownloaded = bytesDownloaded,
                                )
                            )

                            try {
                                downloadQueueDao.updateStatus(item.id, "DOWNLOADING")

                                val result = if (item.youtubeUrl != null) {
                                    // Direct YouTube URL available
                                    downloadManager.downloadByUrl(
                                        url = item.youtubeUrl,
                                        trackId = item.trackId,
                                    )
                                } else {
                                    // Need to search via yt-dlp
                                    downloadManager.downloadBySearch(
                                        query = item.searchQuery,
                                        trackId = item.trackId,
                                    )
                                }

                                if (result.isSuccess) {
                                    downloadQueueDao.updateStatus(item.id, "COMPLETED")
                                    downloadQueueDao.updateCompletedAt(item.id, System.currentTimeMillis())
                                    trackDao.markDownloaded(
                                        item.trackId,
                                        result.filePath,
                                        result.fileSizeBytes,
                                        result.qualityKbps,
                                    )
                                    synchronized(this@TrackDownloadWorker) {
                                        tracksCompleted++
                                        bytesDownloaded += result.fileSizeBytes
                                    }
                                    true
                                } else {
                                    handleDownloadFailure(item, result.error)
                                    synchronized(this@TrackDownloadWorker) {
                                        tracksFailed++
                                    }
                                    false
                                }
                            } catch (e: Exception) {
                                handleDownloadFailure(item, e.message)
                                synchronized(this@TrackDownloadWorker) {
                                    tracksFailed++
                                }
                                false
                            }
                        }
                    }
                }
                results.awaitAll()
            }

            syncStateManager.updateState(
                SyncState.Downloading(1f, "", "", tracksCompleted, total, bytesDownloaded)
            )

            return Result.success(
                workDataOf(
                    PlaylistFetchWorker.KEY_SYNC_ID to syncId,
                    "tracks_downloaded" to tracksCompleted,
                    "tracks_failed" to tracksFailed,
                    "bytes_downloaded" to bytesDownloaded,
                )
            )
        } catch (e: Exception) {
            syncStateManager.updateState(
                SyncState.Error(
                    message = e.message ?: "Download failed",
                    phase = "download",
                    isRetryable = true,
                )
            )
            return Result.failure(workDataOf(PlaylistFetchWorker.KEY_SYNC_ID to syncId))
        }
    }

    private suspend fun handleDownloadFailure(item: com.stash.core.data.db.entity.DownloadQueueEntity, error: String?) {
        val newRetryCount = item.retryCount + 1
        if (newRetryCount < 3) {
            downloadQueueDao.updateForRetry(item.id, newRetryCount, error)
        } else {
            downloadQueueDao.updateStatus(item.id, "FAILED")
            downloadQueueDao.updateError(item.id, error ?: "Max retries exceeded")
        }
    }

    private fun createForegroundInfo(text: String, progress: Int): ForegroundInfo {
        val notification = notificationManager.buildProgressNotification(
            title = "Syncing music",
            text = text,
            progress = progress,
        )
        return if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
            ForegroundInfo(
                SyncNotificationManager.NOTIFICATION_ID_SYNC_PROGRESS,
                notification,
                ServiceInfo.FOREGROUND_SERVICE_TYPE_DATA_SYNC,
            )
        } else {
            ForegroundInfo(
                SyncNotificationManager.NOTIFICATION_ID_SYNC_PROGRESS,
                notification,
            )
        }
    }
}
```

## Task 10: Create SyncFinalizeWorker

**File:** `core/data/src/main/kotlin/com/stash/core/data/sync/workers/SyncFinalizeWorker.kt`

**Why:** Last worker in the chain. Updates sync history, cleans up snapshots, posts summary notification, reschedules the next sync.

```kotlin
package com.stash.core.data.sync.workers

import android.content.Context
import androidx.hilt.work.HiltWorker
import androidx.work.CoroutineWorker
import androidx.work.WorkerParameters
import com.stash.core.data.db.dao.RemoteSnapshotDao
import com.stash.core.data.db.dao.SyncHistoryDao
import com.stash.core.data.sync.SyncNotificationManager
import com.stash.core.data.sync.SyncScheduler
import com.stash.core.data.sync.SyncState
import com.stash.core.data.sync.SyncStateManager
import dagger.assisted.Assisted
import dagger.assisted.AssistedInject

@HiltWorker
class SyncFinalizeWorker @AssistedInject constructor(
    @Assisted appContext: Context,
    @Assisted workerParams: WorkerParameters,
    private val syncHistoryDao: SyncHistoryDao,
    private val snapshotDao: RemoteSnapshotDao,
    private val syncScheduler: SyncScheduler,
    private val syncStateManager: SyncStateManager,
    private val notificationManager: SyncNotificationManager,
) : CoroutineWorker(appContext, workerParams) {

    override suspend fun doWork(): Result {
        val syncId = inputData.getLong(PlaylistFetchWorker.KEY_SYNC_ID, -1L)
        if (syncId == -1L) return Result.failure()

        try {
            syncStateManager.updateState(SyncState.Finalizing(0f))

            val tracksDownloaded = inputData.getInt("tracks_downloaded", 0)
            val tracksFailed = inputData.getInt("tracks_failed", 0)
            val bytesDownloaded = inputData.getLong("bytes_downloaded", 0L)
            val playlistsChecked = inputData.getInt("playlists_checked", 0)
            val newTracksFound = inputData.getInt("new_tracks_found", 0)

            // Update sync history with final results
            syncHistoryDao.updateSyncResult(
                syncId = syncId,
                status = if (tracksFailed == 0) "COMPLETED" else "COMPLETED_WITH_ERRORS",
                playlistsChecked = playlistsChecked,
                newTracksFound = newTracksFound,
                tracksDownloaded = tracksDownloaded,
                tracksFailed = tracksFailed,
                bytesDownloaded = bytesDownloaded,
            )

            syncStateManager.updateState(SyncState.Finalizing(0.5f))

            // Clean up remote snapshots (no longer needed)
            snapshotDao.deleteTrackSnapshots(syncId)
            snapshotDao.deletePlaylistSnapshots(syncId)

            // Prune any orphaned old snapshots (from crashed syncs)
            val oneDayAgo = System.currentTimeMillis() - 86_400_000L
            snapshotDao.pruneOldSnapshots(oneDayAgo)
            snapshotDao.pruneOrphanedTracks()

            syncStateManager.updateState(SyncState.Finalizing(0.8f))

            // Post summary notification
            notificationManager.cancelProgress()
            notificationManager.showSummary(
                newTracks = tracksDownloaded,
                playlistsChecked = playlistsChecked,
                failed = tracksFailed,
            )

            // Reschedule next sync at user-configured time
            syncScheduler.scheduleNextSync()

            syncStateManager.updateState(
                SyncState.Completed(
                    newTracksDownloaded = tracksDownloaded,
                    playlistsChecked = playlistsChecked,
                    tracksFailed = tracksFailed,
                )
            )

            return Result.success()
        } catch (e: Exception) {
            syncStateManager.updateState(
                SyncState.Error(
                    message = e.message ?: "Finalization failed",
                    phase = "finalize",
                    isRetryable = false,
                )
            )
            // Still reschedule even if finalization has issues
            try { syncScheduler.scheduleNextSync() } catch (_: Exception) {}
            return Result.failure()
        }
    }
}
```

## Task 11: Create SyncScheduler -- WorkManager chain orchestration

**File:** `core/data/src/main/kotlin/com/stash/core/data/sync/SyncScheduler.kt`

**Why:** Single entry point for triggering sync (manual or scheduled). Builds the WorkManager chain, computes delay for next scheduled sync, handles constraints.

```kotlin
package com.stash.core.data.sync

import android.content.Context
import androidx.datastore.preferences.core.booleanPreferencesKey
import androidx.datastore.preferences.core.intPreferencesKey
import androidx.work.Constraints
import androidx.work.ExistingWorkPolicy
import androidx.work.NetworkType
import androidx.work.OneTimeWorkRequestBuilder
import androidx.work.WorkManager
import androidx.work.workDataOf
import com.stash.core.data.sync.workers.DiffWorker
import com.stash.core.data.sync.workers.PlaylistFetchWorker
import com.stash.core.data.sync.workers.SyncFinalizeWorker
import com.stash.core.data.sync.workers.TrackDownloadWorker
import dagger.hilt.android.qualifiers.ApplicationContext
import java.util.Calendar
import java.util.concurrent.TimeUnit
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class SyncScheduler @Inject constructor(
    @ApplicationContext private val context: Context,
    private val preferencesManager: SyncPreferencesManager,
) {

    companion object {
        const val UNIQUE_WORK_NAME = "stash_sync"
        const val UNIQUE_SCHEDULED_WORK_NAME = "stash_scheduled_sync"
    }

    private val workManager = WorkManager.getInstance(context)

    /**
     * Trigger an immediate manual sync. Cancels any pending scheduled sync
     * and replaces it (REPLACE policy).
     */
    fun triggerManualSync() {
        val chain = buildSyncChain(trigger = "manual")
        workManager.beginUniqueWork(
            UNIQUE_WORK_NAME,
            ExistingWorkPolicy.REPLACE,
            chain.first,
        )
            .then(chain.second)
            .then(chain.third)
            .then(chain.fourth)
            .enqueue()
    }

    /**
     * Schedule next sync at user-configured time. Uses OneTimeWorkRequest
     * with initialDelay (not PeriodicWorkRequest) for precise timing and chaining support.
     */
    fun scheduleNextSync() {
        val prefs = preferencesManager.getSyncPreferencesBlocking()
        if (!prefs.autoSyncEnabled) return

        val delayMs = computeDelayToNextSync(prefs.syncHour, prefs.syncMinute)

        val constraints = Constraints.Builder()
            .setRequiredNetworkType(
                if (prefs.wifiOnly) NetworkType.UNMETERED else NetworkType.CONNECTED
            )
            .setRequiresBatteryNotLow(true)
            .setRequiresStorageNotLow(true)
            .build()

        val fetchRequest = OneTimeWorkRequestBuilder<PlaylistFetchWorker>()
            .setInputData(workDataOf(PlaylistFetchWorker.KEY_TRIGGER to "scheduled"))
            .setConstraints(constraints)
            .setInitialDelay(delayMs, TimeUnit.MILLISECONDS)
            .addTag("stash_sync")
            .build()

        val diffRequest = OneTimeWorkRequestBuilder<DiffWorker>()
            .setConstraints(constraints)
            .addTag("stash_sync")
            .build()

        val downloadRequest = OneTimeWorkRequestBuilder<TrackDownloadWorker>()
            .setConstraints(constraints)
            .addTag("stash_sync")
            .build()

        val finalizeRequest = OneTimeWorkRequestBuilder<SyncFinalizeWorker>()
            .addTag("stash_sync")
            .build()

        workManager.beginUniqueWork(
            UNIQUE_SCHEDULED_WORK_NAME,
            ExistingWorkPolicy.REPLACE,
            fetchRequest,
        )
            .then(diffRequest)
            .then(downloadRequest)
            .then(finalizeRequest)
            .enqueue()
    }

    /**
     * Cancel all pending sync work.
     */
    fun cancelSync() {
        workManager.cancelUniqueWork(UNIQUE_WORK_NAME)
        workManager.cancelUniqueWork(UNIQUE_SCHEDULED_WORK_NAME)
    }

    /**
     * Compute milliseconds from now until the next occurrence of target hour:minute.
     * If the target time has already passed today, schedule for tomorrow.
     */
    internal fun computeDelayToNextSync(hour: Int, minute: Int): Long {
        val now = Calendar.getInstance()
        val target = Calendar.getInstance().apply {
            set(Calendar.HOUR_OF_DAY, hour)
            set(Calendar.MINUTE, minute)
            set(Calendar.SECOND, 0)
            set(Calendar.MILLISECOND, 0)
        }

        if (target.timeInMillis <= now.timeInMillis) {
            // Target time already passed today, schedule for tomorrow
            target.add(Calendar.DAY_OF_YEAR, 1)
        }

        return target.timeInMillis - now.timeInMillis
    }

    private fun buildSyncChain(trigger: String): SyncChain {
        val constraints = Constraints.Builder()
            .setRequiredNetworkType(NetworkType.UNMETERED)
            .setRequiresBatteryNotLow(true)
            .setRequiresStorageNotLow(true)
            .build()

        val fetchRequest = OneTimeWorkRequestBuilder<PlaylistFetchWorker>()
            .setInputData(workDataOf(PlaylistFetchWorker.KEY_TRIGGER to trigger))
            .setConstraints(constraints)
            .addTag("stash_sync")
            .build()

        val diffRequest = OneTimeWorkRequestBuilder<DiffWorker>()
            .setConstraints(constraints)
            .addTag("stash_sync")
            .build()

        val downloadRequest = OneTimeWorkRequestBuilder<TrackDownloadWorker>()
            .setConstraints(constraints)
            .addTag("stash_sync")
            .build()

        val finalizeRequest = OneTimeWorkRequestBuilder<SyncFinalizeWorker>()
            .addTag("stash_sync")
            .build()

        return SyncChain(fetchRequest, diffRequest, downloadRequest, finalizeRequest)
    }

    private data class SyncChain(
        val first: androidx.work.OneTimeWorkRequest,
        val second: androidx.work.OneTimeWorkRequest,
        val third: androidx.work.OneTimeWorkRequest,
        val fourth: androidx.work.OneTimeWorkRequest,
    )
}
```

```kotlin
package com.stash.core.data.sync

import android.content.Context
import androidx.datastore.preferences.core.booleanPreferencesKey
import androidx.datastore.preferences.core.edit
import androidx.datastore.preferences.core.intPreferencesKey
import androidx.datastore.preferences.preferencesDataStore
import dagger.hilt.android.qualifiers.ApplicationContext
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.first
import kotlinx.coroutines.flow.map
import kotlinx.coroutines.runBlocking
import javax.inject.Inject
import javax.inject.Singleton

private val Context.syncDataStore by preferencesDataStore(name = "sync_preferences")

data class SyncPreferences(
    val autoSyncEnabled: Boolean = true,
    val syncHour: Int = 3, // Default 3:00 AM
    val syncMinute: Int = 0,
    val wifiOnly: Boolean = true,
)

@Singleton
class SyncPreferencesManager @Inject constructor(
    @ApplicationContext private val context: Context,
) {
    companion object {
        val KEY_AUTO_SYNC_ENABLED = booleanPreferencesKey("auto_sync_enabled")
        val KEY_SYNC_HOUR = intPreferencesKey("sync_hour")
        val KEY_SYNC_MINUTE = intPreferencesKey("sync_minute")
        val KEY_WIFI_ONLY = booleanPreferencesKey("wifi_only")
    }

    val syncPreferencesFlow: Flow<SyncPreferences> = context.syncDataStore.data.map { prefs ->
        SyncPreferences(
            autoSyncEnabled = prefs[KEY_AUTO_SYNC_ENABLED] ?: true,
            syncHour = prefs[KEY_SYNC_HOUR] ?: 3,
            syncMinute = prefs[KEY_SYNC_MINUTE] ?: 0,
            wifiOnly = prefs[KEY_WIFI_ONLY] ?: true,
        )
    }

    /**
     * Blocking read for use in WorkManager workers (which don't have a coroutine scope
     * available at construction time). Use sparingly.
     */
    fun getSyncPreferencesBlocking(): SyncPreferences = runBlocking {
        syncPreferencesFlow.first()
    }

    suspend fun updateAutoSync(enabled: Boolean) {
        context.syncDataStore.edit { it[KEY_AUTO_SYNC_ENABLED] = enabled }
    }

    suspend fun updateSyncTime(hour: Int, minute: Int) {
        context.syncDataStore.edit {
            it[KEY_SYNC_HOUR] = hour
            it[KEY_SYNC_MINUTE] = minute
        }
    }

    suspend fun updateWifiOnly(wifiOnly: Boolean) {
        context.syncDataStore.edit { it[KEY_WIFI_ONLY] = wifiOnly }
    }
}
```

## Task 13: Create BootReceiver for rescheduling sync after reboot

**File:** `app/src/main/kotlin/com/stash/app/receiver/BootReceiver.kt`

```kotlin
package com.stash.app.receiver

import android.content.BroadcastReceiver
import android.content.Context
import android.content.Intent
import com.stash.core.data.sync.SyncScheduler
import dagger.hilt.android.AndroidEntryPoint
import javax.inject.Inject

@AndroidEntryPoint
class BootReceiver : BroadcastReceiver() {

    @Inject
    lateinit var syncScheduler: SyncScheduler

    override fun onReceive(context: Context, intent: Intent) {
        if (intent.action == Intent.ACTION_BOOT_COMPLETED ||
            intent.action == Intent.ACTION_MY_PACKAGE_REPLACED
        ) {
            // WorkManager persists work in SQLite and will re-enqueue automatically,
            // but our self-rescheduling approach means we should explicitly reschedule
            // in case the last SyncFinalizeWorker hadn't run before shutdown.
            syncScheduler.scheduleNextSync()
        }
    }
}
```

## Task 14: Create OemBatteryHelper for aggressive OEM detection

**File:** `core/common/src/main/kotlin/com/stash/core/common/OemBatteryHelper.kt`

**Why:** Xiaomi, Huawei, Samsung, OnePlus, Oppo, and Vivo all have aggressive battery killers that will murder background WorkManager jobs. The user must be guided to whitelist the app.

```kotlin
package com.stash.core.common

import android.content.Context
import android.content.Intent
import android.net.Uri
import android.os.Build

data class OemBatteryInfo(
    val manufacturer: String,
    val requiresUserAction: Boolean,
    val settingsIntent: Intent?,
    val dontkillmyappUrl: String,
    val instructions: String,
)

object OemBatteryHelper {

    private val AGGRESSIVE_OEMS = setOf(
        "xiaomi", "redmi", "poco",
        "huawei", "honor",
        "samsung",
        "oneplus",
        "oppo", "realme",
        "vivo", "iqoo",
        "meizu",
        "asus",
        "nokia", "hmd global",
        "sony",
    )

    fun isAggressiveOem(): Boolean {
        return Build.MANUFACTURER.lowercase() in AGGRESSIVE_OEMS
    }

    fun getOemBatteryInfo(): OemBatteryInfo? {
        val manufacturer = Build.MANUFACTURER.lowercase()
        if (manufacturer !in AGGRESSIVE_OEMS) return null

        val dontkillmyappUrl = "https://dontkillmyapp.com/${
            when {
                manufacturer in setOf("xiaomi", "redmi", "poco") -> "xiaomi"
                manufacturer in setOf("huawei", "honor") -> "huawei"
                manufacturer == "samsung" -> "samsung"
                manufacturer == "oneplus" -> "oneplus"
                manufacturer in setOf("oppo", "realme") -> "oppo"
                manufacturer in setOf("vivo", "iqoo") -> "vivo"
                else -> manufacturer
            }
        }"

        val settingsIntent: Intent? = when {
            manufacturer in setOf("xiaomi", "redmi", "poco") -> {
                Intent().apply {
                    setClassName(
                        "com.miui.powerkeeper",
                        "com.miui.powerkeeper.ui.HiddenAppsConfigActivity",
                    )
                    addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
                }
            }
            manufacturer == "samsung" -> {
                Intent().apply {
                    setClassName(
                        "com.samsung.android.lool",
                        "com.samsung.android.sm.battery.ui.BatteryActivity",
                    )
                    addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
                }
            }
            manufacturer in setOf("huawei", "honor") -> {
                Intent().apply {
                    setClassName(
                        "com.huawei.systemmanager",
                        "com.huawei.systemmanager.startupmgr.ui.StartupNormalAppListActivity",
                    )
                    addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
                }
            }
            else -> null
        }

        val instructions = when {
            manufacturer in setOf("xiaomi", "redmi", "poco") ->
                "Go to Settings > Apps > Manage apps > Stash > Battery saver > No restrictions"
            manufacturer == "samsung" ->
                "Go to Settings > Battery > Background usage limits > Never sleeping apps > Add Stash"
            manufacturer in setOf("huawei", "honor") ->
                "Go to Settings > Battery > App launch > Stash > Manage manually > Enable all toggles"
            manufacturer == "oneplus" ->
                "Go to Settings > Battery > Battery optimization > Stash > Don't optimize"
            else ->
                "Visit dontkillmyapp.com for instructions specific to your device"
        }

        return OemBatteryInfo(
            manufacturer = Build.MANUFACTURER,
            requiresUserAction = true,
            settingsIntent = settingsIntent,
            dontkillmyappUrl = dontkillmyappUrl,
            instructions = instructions,
        )
    }

    fun openDontkillmyapp(context: Context) {
        val info = getOemBatteryInfo() ?: return
        val intent = Intent(Intent.ACTION_VIEW, Uri.parse(info.dontkillmyappUrl))
        intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
        context.startActivity(intent)
    }

    fun tryOpenOemSettings(context: Context): Boolean {
        val info = getOemBatteryInfo() ?: return false
        val intent = info.settingsIntent ?: return false
        return try {
            context.startActivity(intent)
            true
        } catch (e: Exception) {
            false
        }
    }
}
```

```kotlin
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.SharingStarted
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.combine
import kotlinx.coroutines.flow.stateIn
import kotlinx.coroutines.launch
import javax.inject.Inject

data class SyncUiState(
    val syncState: SyncState = SyncState.Idle,
    val overallProgress: Float = 0f,
    val preferences: SyncPreferences = SyncPreferences(),
    val syncHistory: List<SyncHistoryEntity> = emptyList(),
    val isSpotifyConnected: Boolean = false,
    val isYouTubeConnected: Boolean = false,
    val showOemBatteryWarning: Boolean = false,
    val oemBatteryInstructions: String? = null,
    val showTimePickerDialog: Boolean = false,

@HiltViewModel
class SyncViewModel @Inject constructor(
    private val syncScheduler: SyncScheduler,
    private val syncStateManager: SyncStateManager,
    private val syncPreferencesManager: SyncPreferencesManager,
    private val syncHistoryDao: SyncHistoryDao,
    private val tokenManager: TokenManager,
) : ViewModel() {

    private val _showTimePicker = MutableStateFlow(false)

    val uiState: StateFlow<SyncUiState> = combine(
        syncStateManager.state,
        syncPreferencesManager.syncPreferencesFlow,
        syncHistoryDao.getRecentSyncs(20),
        _showTimePicker,
    ) { syncState, preferences, history, showTimePicker ->
        SyncUiState(
            syncState = syncState,
            overallProgress = syncStateManager.overallProgress(),
            preferences = preferences,
            syncHistory = history,
            isSpotifyConnected = tokenManager.hasSpotifyToken(),
            isYouTubeConnected = tokenManager.hasYouTubeToken(),
            showOemBatteryWarning = OemBatteryHelper.isAggressiveOem(),
            oemBatteryInstructions = OemBatteryHelper.getOemBatteryInfo()?.instructions,
            showTimePickerDialog = showTimePicker,
        )
    }.stateIn(
        viewModelScope,
        SharingStarted.WhileSubscribed(5000),
        SyncUiState(),
    )

    fun triggerManualSync() {
        syncScheduler.triggerManualSync()
    }

    fun cancelSync() {
        syncScheduler.cancelSync()
        syncStateManager.reset()
    }

    fun setAutoSync(enabled: Boolean) {
        viewModelScope.launch {
            syncPreferencesManager.updateAutoSync(enabled)
            if (enabled) {
                syncScheduler.scheduleNextSync()
            } else {
                syncScheduler.cancelSync()
            }
        }
    }

    fun setWifiOnly(wifiOnly: Boolean) {
        viewModelScope.launch {
            syncPreferencesManager.updateWifiOnly(wifiOnly)
            // Reschedule with new constraints
            syncScheduler.scheduleNextSync()
        }
    }

    fun showTimePicker() {
        _showTimePicker.value = true
    }

    fun dismissTimePicker() {
        _showTimePicker.value = false
    }

    fun setSyncTime(hour: Int, minute: Int) {
        viewModelScope.launch {
            syncPreferencesManager.updateSyncTime(hour, minute)
            _showTimePicker.value = false
            // Reschedule at new time
            syncScheduler.scheduleNextSync()
        }
    }
}
```

3. **Connect Spotify** -- Source shows green dot and checkmark
4. **Tap Sync Now** -- Notification permission requested (API 33+), sync starts
5. **Progress updates** -- State cycles: Authenticating -> Fetching -> Diffing -> Downloading -> Finalizing -> Completed
6. **Notification** -- Foreground notification shows track-by-track progress
7. **Summary notification** -- Posted on completion: "Added N tracks from M playlists"
8. **Sync history** -- New entry appears in history list
9. **Schedule toggle** -- Enable auto-sync, set time, verify via `adb shell dumpsys jobscheduler | grep stash`
10. **Reboot device** -- Verify sync is rescheduled (check WorkManager unique work via `adb shell dumpsys jobscheduler`)
11. **OEM warning** -- If on Xiaomi/Samsung/etc., warning card with instructions appears
12. **Cancel sync** -- Mid-sync cancel stops workers and resets state to Idle
13. **WiFi constraint** -- Disconnect WiFi, verify sync pauses; reconnect, verify it resumes

### Critical Files for Implementation

- `core/data/src/main/kotlin/com/stash/core/data/sync/workers/DiffWorker.kt` - Most complex logic: snapshot_id incremental sync, cross-platform dedup via TrackMatcher, download queue creation
- `core/data/src/main/kotlin/com/stash/core/data/sync/SyncScheduler.kt` - Orchestrates the entire WorkManager chain, handles scheduling, manual triggers, and rescheduling
- `core/data/src/main/kotlin/com/stash/core/data/sync/workers/TrackDownloadWorker.kt` - Foreground service promotion, concurrent download semaphore, longest-running phase (70% of sync)
- `core/data/src/main/kotlin/com/stash/core/data/sync/SyncStateManager.kt` - Central state machine observed by all workers, notifications, and UI; must be thread-safe
- `feature/sync/src/main/kotlin/com/stash/feature/sync/SyncScreen.kt` - Full sync UI: sources, progress, schedule config, history log, OEM battery warnings
